## Sample Juniper JunOS config (simplified for testing)
## This would be a real customer config in production

system {
    host-name jnpr-tor1;
    root-authentication {
        encrypted-password "***";
    }
    login {
        user admin {
            class super-user;
        }
    }
    services {
        ssh;
        netconf {
            ssh;
        }
    }
}

interfaces {
    xe-0/0/0 {
        description "Host Interface 1";
        unit 0 {
            family ethernet-switching {
                interface-mode trunk;
                vlan members [ vlan100 vlan200 vlan711 ];
            }
        }
    }
    xe-0/0/1 {
        description "Host Interface 2";
        unit 0 {
            family ethernet-switching {
                interface-mode trunk;
                vlan members [ vlan100 vlan200 vlan712 ];
            }
        }
    }
    et-0/0/48 {
        description "Uplink to Spine";
        unit 0 {
            family inet {
                address 10.0.0.1/30;
            }
        }
    }
}

vlans {
    vlan100 {
        description "Management";
        vlan-id 100;
        l3-interface irb.100;
    }
    vlan200 {
        description "Compute";
        vlan-id 200;
    }
    vlan711 {
        description "Storage-1";
        vlan-id 711;
    }
    vlan712 {
        description "Storage-2";
        vlan-id 712;
    }
}

routing-options {
    static {
        route 0.0.0.0/0 next-hop 10.0.0.2;
    }
}

protocols {
    bgp {
        group fabric {
            type external;
            local-as 65001;
            neighbor 10.0.0.2 {
                peer-as 65000;
            }
        }
    }
    lldp {
        interface all;
    }
}
